Eg. Code

    var easter = $(window).easteregg();
    ghost_counter = 0;
    easter.addEgg('demo', function(){
        $('body').append('<img id="ghost-pixel-' + ghost_counter + '" src="./img/ghost.png" />');
        var $ghost = $('#ghost-pixel-' + ghost_counter);
        $ghost.css('position', 'absolute');
        $ghost.css('top', getRandomInt(0, window.innerHeight));
        $ghost.css('left', getRandomInt(0, window.innerWidth));
        $ghost.css('z-index', 9999 + ghost_counter);
        ghost_counter++;
    });
    easter.init();
